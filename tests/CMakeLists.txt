# tests/CMakeLists.txt\n# Integration tests for the rastertokpsl CUPS filter pipeline.\n# These are end-to-end tests that exercise the full PDF -> raster -> KPSL\n# conversion chain using the open-source driver built from src/.\n# -----------------------------------------------------------------------------\n\ncmake_minimum_required(VERSION 3.25)\n\nset(TEST_RUNTIME_DIR \"$<TARGET_FILE_DIR:test_raster>\")\n\n# -----------------------------------------------------------------------------\n# raster_chain: PDF -> CUPS raster -> KPSL (direct filter invocation)\n# -----------------------------------------------------------------------------\nadd_executable(test_raster test_raster.cxx)\nset_target_properties(test_raster\n    PROPERTIES\n        CXX_STANDARD 26\n        CXX_STANDARD_REQUIRED ON\n        CXX_EXTENSIONS OFF\n)\n\nadd_dependencies(test_raster rastertokpsl)\n\nadd_custom_command(\n    TARGET test_raster POST_BUILD\n    COMMAND ${CMAKE_COMMAND} -E create_symlink\n        $<TARGET_FILE:rastertokpsl>\n        ${TEST_RUNTIME_DIR}/rastertokpsl\n    COMMENT \"Linking rastertokpsl into test runtime directory\"\n)\n\nadd_custom_command(\n    TARGET test_raster POST_BUILD\n    COMMAND ${CMAKE_COMMAND} -E create_symlink\n        \"${CMAKE_CURRENT_SOURCE_DIR}/test.pdf\"\n        ${TEST_RUNTIME_DIR}/test.pdf\n    COMMENT \"Linking test.pdf into test runtime directory\"\n)\n\nadd_test(\n    NAME raster_chain\n    COMMAND $<TARGET_FILE:test_raster>\n    WORKING_DIRECTORY ${TEST_RUNTIME_DIR}\n)\n\nset_tests_properties(raster_chain\n    PROPERTIES\n        TIMEOUT 60\n        LABELS \"integration;filter\"\n)\n\n# -----------------------------------------------------------------------------\n# filter_chain: Full CUPS pipeline via cupsfilter + PPD\n# -----------------------------------------------------------------------------\nadd_executable(test_filter test_filter.cxx)\nset_target_properties(test_filter\n    PROPERTIES\n        CXX_STANDARD 26\n        CXX_STANDARD_REQUIRED ON\n        CXX_EXTENSIONS OFF\n)\n\nadd_dependencies(test_filter rastertokpsl)\n\nadd_custom_command(\n    TARGET test_filter POST_BUILD\n    COMMAND ${CMAKE_COMMAND} -E create_symlink\n        \"${CMAKE_CURRENT_SOURCE_DIR}/../ppd/English/Kyocera_FS-1020MFPGDI.ppd\"\n        ${TEST_RUNTIME_DIR}/Kyocera_FS-1020MFPGDI.ppd\n    COMMENT \"Linking Kyocera_FS-1020MFPGDI.ppd into test runtime directory\"\n)\n\nadd_custom_command(\n    TARGET test_filter POST_BUILD\n    COMMAND ${CMAKE_COMMAND} -E create_symlink\n        \"${CMAKE_CURRENT_SOURCE_DIR}/test.pdf\"\n        ${TEST_RUNTIME_DIR}/test.pdf\n    COMMENT \"Linking test.pdf into test runtime directory\"\n)\n\nadd_test(\n    NAME filter_chain\n    COMMAND $<TARGET_FILE:test_filter>\n    WORKING_DIRECTORY ${TEST_RUNTIME_DIR}\n)\n\nset_tests_properties(filter_chain\n    PROPERTIES\n        TIMEOUT 60\n        LABELS \"integration;cups\"\n)\n
